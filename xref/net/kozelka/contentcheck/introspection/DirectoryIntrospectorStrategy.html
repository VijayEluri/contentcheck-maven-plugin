12  /**
13   * Implementation of {@link IntrospectorInputStrategy} which can read the content of ordinary directory.
14   */
15  class DirectoryIntrospectorStrategy implements IntrospectorInputStrategy {
16  
17      public Set<String> list(File baseDirectory) throws IOException {
18          if (baseDirectory == null) {
19              throw new IllegalArgumentException("containerFile cannot be null!");
20          };
21          if (!baseDirectory.isDirectory()) {
22              throw new IllegalArgumentException(baseDirectory.getAbsolutePath() + " is not a directory!");
23          }
24          final Set<String> entries = new HashSet<String>();
25          entries.addAll(FileUtils.getFileNames(baseDirectory, null, null, false));
26          final List<String> directories = FileUtils.getDirectoryNames(baseDirectory, null, null, false);
27          // strangely, the empty string is returned in result, we have to remove it explicitly.
28          directories.remove("");
29          // directories does not have trailing slash, so we need to add it
30          for (String directory : directories) {
31              entries.add(directory + "/");
32          }
33          return entries;
34      }
35  
36      public InputStream getInputStream(File containerFile, String entry) throws IOException {
37          return new FileInputStream(new File(containerFile, entry));
38      }
