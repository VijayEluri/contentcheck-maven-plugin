42      /**
43       * Starts reading {@code sourceFile}'s content entry by entry. If an entry matches {@link #checkFilesPattern}
44       * and is not a vendor archive (in case of {@link #ignoreVendorArchives} is <code>true</code>)
45       * the entry will be delegated to the method {@link #processEntry(String)}
46       * for further processing.
47       * 
48       * @param sourceFile a source file to be read, typically an archive or directory
49       * 
50       * @return the number of read entities.
51       * 
52       * @see #processEntry(String)
53       */
54      public final int readEntries(final File sourceFile) throws IOException {
55          log.info("Reading source file: " + sourceFile);
56          final IntrospectorInputStrategy inputStrategy;
57          if (sourceFile.isDirectory()) {
58              inputStrategy = new DirectoryIntrospectorStrategy();
59          } else {
60              inputStrategy = new ZipArchiveIntrospectorStrategy();
61          }
62  
63          int totalCnt = 0;
64          for (String entry : inputStrategy.readAllEntries(sourceFile)) {
65              totalCnt++;
66              if (!shouldBeChecked(entry)) {
67                  log.debug(String.format("Skipping entry '%s' doesn't match with pattern '%s'", entry, checkFilesPattern));
68                  continue;
69              }
70  
71              if(isJarFileExtension(entry) && ignoreVendorArchives) {
72                  //
73                  if(isVendorArchive(entry, inputStrategy.readEntryData(sourceFile, entry))) {
74                      log.debug(String.format("Skipping entry '%s' it's a vendor archive", entry));
75                      continue;
76                  }
77              }
78  
79              processEntry(entry);
80          }
81  
82          return totalCnt;
83      }

140     private boolean isVendorArchive(final String entryName, final InputStream archiveEntryData) throws IOException {
141         try {
142             final File tempFile = copyStreamToTemporaryFile(entryName, archiveEntryData);
143             tempFile.deleteOnExit();
144             final boolean vendorArchive = checkArchiveManifest(entryName, tempFile);
145             tempFile.delete();//only for sure if the plugin is used in long live JVM
146             return vendorArchive;
147         } finally {
148             archiveEntryData.close();
149         }
150     }
151 
152     /**
153      * @return true when vendorId matches with jar's manifest otherwise false
154      */
155     private boolean checkArchiveManifest(final String jarPath, File tempJAR) {
156         JarFile jarFile = null;
157         try {
158             jarFile = new JarFile(tempJAR);
159             final Manifest manifest = jarFile.getManifest();
160             if(manifest != null) {
161                 final Attributes mainAttributes = manifest.getMainAttributes();
162                 if(mainAttributes != null) {
163                     final String vendor = mainAttributes.getValue(manifestVendorEntry);
164                     return vendorId.equals(vendor);
165                 }
166             }
167 
168         } catch (IOException e) {
169             log.warn("Cannot check MANIFEST.MF file in JAR archive " + jarPath, e);
170         } finally {
171             if(jarFile != null) {
172                 try {
173                     jarFile.close();
174                 } catch (IOException e) {
175                     log.warn("Cannot close temporary JAR file " + jarPath,e);
176                 }
177             }
178         }
179         return false;
180     }

182     private File copyStreamToTemporaryFile(final String jarPath, final InputStream archiveInputStream) throws IOException {
183         final File tempFile = File.createTempFile(UUID.randomUUID().toString(), "jar");
184         final FileOutputStream fos = new  FileOutputStream(tempFile);
185         try {
186             log.debug("Checking " + jarPath + " to be a vendor archive, using tempfile " + tempFile);
187             IOUtil.copy(archiveInputStream, fos);
188             return tempFile;
189         } finally {
190             fos.close();
191         }
192     }
