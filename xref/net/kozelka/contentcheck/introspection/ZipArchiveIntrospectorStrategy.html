16  class ZipArchiveIntrospectorStrategy implements IntrospectorInputStrategy {
17      public Set<String> list(File containerFile) throws IOException {
18          final ZipFile zipFile = new ZipFile(containerFile);
19          final ZipInputStream zis = new ZipInputStream(new FileInputStream(containerFile));
20          final Set<String> entries = new HashSet<String>();
21          try {
22              ZipEntry entry;
23              while ((entry = zis.getNextEntry()) != null) {
24                  final String entryName = entry.getName();
25                  entries.add(entryName);
26              }
27          } finally {
28              try {
29                  zis.close();
30              } finally {
31                  zipFile.close();
32              }
33          }
34  
35          return entries;
36      }
37  
38      public InputStream getInputStream(File containerFile, String entry) throws IOException {
39          return new ZipFile(containerFile).getInputStream(new ZipEntry(entry));
