27          super();

32      public Map<String, List<License>> parseLicenseMappingFile() throws IOException, MojoFailureException{
33          final String licenseMappingFilePath = licenseMappingFile.getPath();
34          final Map<String, Object> json;
35          try {
36              log.info(String.format("Reading license mapping file %s", licenseMappingFilePath));
37              json= objectMapper.readValue(licenseMappingFile, Map.class);
38          } catch (JsonParseException e) {
39              throw new MojoFailureException(String.format("Cannot parse JSON from file %s the content of the file is not well formed JSON.", licenseMappingFilePath),e);
40          } catch (JsonMappingException e) {
41              throw new MojoFailureException(String.format("Cannot deserialize JSON from file %s", licenseMappingFilePath),e);
42          }
43          final Map <String, List<License>> fileToLicenseMapping = new HashMap<String, List<License>>();
44          final List<Map<String, Object>> jsonLicenses = (List<Map<String, Object>>) json.get("licenses");
45          //construct model objects from JSON
46          for (Map<String, Object> jsonLicense : jsonLicenses) {
47              final License license = new License();
48              license.setName((String) jsonLicense.get("name"));
49              license.setUrl((String) jsonLicense.get("url"));
50              final List<String> fileNames = (List<String>) jsonLicense.get("files");
51              for (String fileName : fileNames) {
52                  fileToLicenseMapping.put(fileName, Arrays.asList(license));
53              }
54          }
55          return fileToLicenseMapping;
56      }
