16  /**
17   * Checks the archive content according to an authoritative source. This authoritative source
18   * defines set of allowed files in the archive.
19   */
20  @Mojo(name = "check", defaultPhase = LifecyclePhase.VERIFY, threadSafe = true)
21  public class ContentCheckMojo extends AbstractArchiveContentMojo {
22  
23      /**
24       * If true, no check is performed.
25       */
26      @Parameter(defaultValue = "false", property = "contentcheck.skip")
27      boolean skip;
28  
29      /**
30       * The file with list of approved files. If such file does not exist, the check is skipped. This enables multimodule use.
31       * Each line in represents one pathname entry.
32       * Empty lines and comments (starting with '#') are ignored.
33       */
34      @Parameter(defaultValue = "${basedir}/approved-content.txt")
35      File contentListing;
36  
37      /**
38       * Message used to report missing entry - uses the {@link java.util.Formatter} syntax to embed entry name.
39       */
40      @Parameter(defaultValue = "File is expected but not found: %s")
41      String msgMissing;
42  
43      /**
44       * Message used to report unexpected entry - uses the {@link java.util.Formatter} syntax to embed entry name.
45       */
46      @Parameter(defaultValue = "Found unexpected file: %s")
47      String msgUnexpected;
48      
49      /**
50       * If true, stops the build when there is any file missing.
51       */
52      @Parameter(defaultValue = "false")
53      boolean failOnMissing;
54  
55      /**
56       * If true, stops the build when there is any unexpected file.
57       */
58      @Parameter(defaultValue = "true")
59      boolean failOnUnexpected;
60  
61      /**
62       * If true, modules with POM packaging are skipped (excluded from the content check).
63       */
64      @Parameter(defaultValue = "false")
65      boolean skipPOMPackaging;
66  
67      @Override
68      public void execute() throws MojoExecutionException, MojoFailureException {
69  
70          if (skip) {
71              getLog().info("Content checking is skipped.");
72              return;
73          }
74  
75          if(!contentListing.exists()) {
76              getLog().info("Skipping - file does not exist: " + contentListing);
77              return;
78          }
79  
80          assertSourceFileExists();
81  
82          try {
83              final MyIntrospectionListener introspectionListener = new MyIntrospectionListener(getLog());
84              final ContentIntrospector introspector = ContentIntrospector.create(introspectionListener, ignoreVendorArchives, vendorId, manifestVendorEntry, checkFilesPattern);
85              introspector.setSourceFile(sourceFile);
86              final ContentChecker contentChecker = new ContentChecker();
87              contentChecker.getEvents().addListener(new MyContentCheckerListener(getLog()));
88              contentChecker.setIntrospector(introspector);
89              //
90              getLog().info("Reading listing: " + contentListing);
91              final CheckerOutput output = contentChecker.check(contentListing);
92  
93              // report missing entries
94              final Set<String> missingEntries = output.diffMissingEntries();
95              for (String entry : missingEntries) {
96                  log(failOnMissing, String.format(msgMissing, entry));
97              }
98              // report unexpected entries
99              final Set<String> unexpectedEntries = output.diffUnexpectedEntries();
100             for (String entry : unexpectedEntries) {
101                 log(failOnUnexpected, String.format(msgUnexpected, entry));
102             }
103             // error summary
104             if (missingEntries.size() > 0) {
105                 log(failOnMissing, "Missing: " + missingEntries.size() + " entries");
106             }
107             if (unexpectedEntries.size() > 0) {
108                 log(failOnUnexpected, "Unexpected: " + unexpectedEntries.size() + " entries");
109             }
110             // fail as neccessary, after reporting all detected problems
111             if (failOnMissing && ! missingEntries.isEmpty()) {
112                 throw new MojoFailureException(missingEntries.size() + " expected entries are missing in " + sourceFile);
113             }
114 
115             if (failOnUnexpected && ! unexpectedEntries.isEmpty()) {
116                 throw new MojoFailureException(unexpectedEntries.size() + " unexpected entries appear in " + sourceFile);
117             }
118 
119             getLog().info("Source " + sourceFile.getAbsolutePath() + " has valid content according to " + contentListing.getAbsolutePath());
120         } catch (IOException e) {
121             throw new MojoFailureException(e.getMessage(), e);
122         }
123     }
124 
125     private void log(boolean error, String message) {
126         if (error) {
127             getLog().error(message);
128         } else {
129             getLog().warn(message);
130         }
131     }
