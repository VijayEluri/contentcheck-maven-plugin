14  /**
15   * Checks the archive content according to an authoritative source. This authoritative source
16   * defines set of allowed files in the archive.
17   */
18  @Mojo(name = "check", defaultPhase = LifecyclePhase.VERIFY, threadSafe = true)
19  public class ContentCheckMojo extends AbstractArchiveContentMojo {
20      
21      /**
22       * Message used to report missing entry - uses the {@link java.util.Formatter} syntax to embed entry name.
23       */
24      @Parameter(defaultValue = "File is expected but not found: %s")
25      String msgMissing;
26  
27      /**
28       * Message used to report unexpected entry - uses the {@link java.util.Formatter} syntax to embed entry name.
29       */
30      @Parameter(defaultValue = "Found unexpected file: %s")
31      String msgUnexpected;
32      
33      /**
34       * If true, stops the build when there is any file missing.
35       */
36      @Parameter(defaultValue = "false")
37      boolean failOnMissing;
38  
39      /**
40       * If true, stops the build when there is any unexpected file.
41       */
42      @Parameter(defaultValue = "true")
43      boolean failOnUnexpected;
44  
45      /**
46       * If true, modules with POM packaging are skipped (excluded from the content check).
47       */
48      @Parameter(defaultValue = "false")
49      boolean skipPOMPackaging;
50  
51      protected void doExecute() throws IOException, MojoExecutionException, MojoFailureException {
52          
53          if(!contentListing.exists()) {
54              getLog().info("Skipping - file does not exist: " + contentListing);
55          }
56  
57          assertSourceFileExists();
58  
59          final ContentChecker contentChecker = new ContentChecker(getLog(), ignoreVendorArchives, vendorId, manifestVendorEntry, checkFilesPattern);
60          final CheckerOutput output = contentChecker.check(contentListing, sourceFile);
61  
62          // report missing entries
63          final Set<String> missingEntries = output.diffMissingEntries();
64          for (String entry : missingEntries) {
65              log(failOnMissing, String.format(msgMissing, entry));
66          }
67          // report unexpected entries
68          final Set<String> unexpectedEntries = output.diffUnexpectedEntries();
69          for (String entry : unexpectedEntries) {
70              log(failOnUnexpected, String.format(msgUnexpected, entry));
71          }
72          // error summary
73          if (missingEntries.size() > 0) {
74              log(failOnMissing, "Missing: " + missingEntries.size() + " entries");
75          }
76          if (unexpectedEntries.size() > 0) {
77              log(failOnUnexpected, "Unexpected: " + unexpectedEntries.size() + " entries");
78          }
79          // fail as neccessary, after reporting all detected problems
80          if (failOnMissing && ! missingEntries.isEmpty()) {
81              throw new MojoFailureException(missingEntries.size() + " expected entries are missing in " + sourceFile);
82          }
83  
84          if (failOnUnexpected && ! unexpectedEntries.isEmpty()) {
85              throw new MojoFailureException(unexpectedEntries.size() + " unexpected entries appear in " + sourceFile);
86          }
87  
88          getLog().info("Source " + sourceFile.getAbsolutePath() + " has valid content according to " + contentListing.getAbsolutePath());
89      }
90  
91      private void log(boolean error, String message) {
92          if (error) {
93              getLog().error(message);
94          } else {
95              getLog().warn(message);
96          }
97      }
