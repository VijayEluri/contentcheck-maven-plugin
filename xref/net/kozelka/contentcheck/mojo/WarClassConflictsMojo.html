21  /**
22   * Looks for conflicts within the libraries in given sourceFile.
23   *
24   * @author Petr Kozelka
25   * @since 1.0.3
26   */
27  @Mojo(name="warcc", defaultPhase = LifecyclePhase.VERIFY)
28  public class WarClassConflictsMojo extends AbstractMojo {
29      /**
30       * If true, no check is performed.
31       */
32      @Parameter(defaultValue = "false", property = "contentcheck.skip")
33      boolean skip;
34  
35      /**
36       * The archive file to be checked
37       */
38      @Parameter(defaultValue = "${project.build.directory}/${project.build.finalName}.war", property = "sourceFile")
39      File sourceFile;
40  
41      /**
42       * How many class conflicts to list directly. Use <code>-1</code> to list all.
43       */
44      @Parameter(defaultValue = "5")
45      int previewThreshold;
46  
47      /**
48       * How many overlaps are we tolerating.
49       * Useful to ensure that the number is not growing, when you cannot fix everything.
50       * @todo replace this with include/exclude lists
51       */
52      @Parameter(defaultValue = "0")
53      int toleratedOverlapCount;
54  
55      /**
56       * @deprecated Use {@link #toleratedOverlapCount} instead.
57       */
58      @Deprecated
59      @Parameter(defaultValue = "-1")
60      int toleratedConflictCount;
61  
62      public void execute() throws MojoExecutionException, MojoFailureException {
63          if (skip) {
64              getLog().info("Content conflict checking is skipped.");
65              return;
66          }
67  
68          if (toleratedConflictCount > -1) {
69              getLog().warn("Parameter 'toleratedConflictCount' is deprecated - use 'toleratedOverlapCount' instead");
70              if (toleratedOverlapCount > 0) {
71                  getLog().warn("Parameter 'toleratedConflictCount' is just a deprecated variant of 'toleratedOverlapCount'. You specified both, using only the latter.");
72              } else {
73                  toleratedOverlapCount = toleratedConflictCount;
74              }
75          }
76          //
77          try {
78              final ClassConflictAnalyzer ccd = new ClassConflictAnalyzer();
79              final List<ArchiveInfo> archives = ArchiveLoader.loadWar(sourceFile);
80              final ClassConflictReport report = ccd.analyze(archives);
81              final List<ArchiveConflict> archiveConflicts = report.getArchiveConflicts();
82              final int totalOverlaps = report.getTotalOverlaps();
83              if (archiveConflicts.isEmpty()) {
84                  getLog().info("No overlaps detected.");
85              } else {
86                  final ClassConflictPrinter printer = new ClassConflictPrinter();
87                  printer.setPreviewThreshold(previewThreshold);
88                  final StreamConsumer output = new StreamConsumer() {
89                      public void consumeLine(String line) {
90                          getLog().error(line);
91                      }
92                  };
93                  printer.setOutput(output);
94                  printer.print(report);
95                  final ConflictingResourcesReport printer2 = new ConflictingResourcesReport();
96                  printer2.setOutput(output);
97                  printer2.print(report);
98                  final String errorMessage = String.format("Found %d overlapping resources in %d archive conflicts in %s",
99                      totalOverlaps,
100                     archiveConflicts.size(),
101                     sourceFile);
102                 getLog().error(errorMessage);
103                 if (totalOverlaps > toleratedOverlapCount) {
104                     throw new MojoFailureException(errorMessage);
105                 }
106             }
107             if (totalOverlaps < toleratedOverlapCount) {
108                 getLog().warn(String.format("We currently tolerate %d overlaps; please reduce the tolerance to prevent growing mess", toleratedOverlapCount));
109             }
110         } catch (IOException e) {
111             throw new MojoExecutionException(sourceFile.getAbsolutePath(), e);
112         }
113     }
