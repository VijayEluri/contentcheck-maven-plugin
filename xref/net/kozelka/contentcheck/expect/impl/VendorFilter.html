15  /**
16   * Determines if an entry is matching given vendor.
17   *
18   * @author Petr Kozelka
19   */
20  public class VendorFilter implements ContentIntrospector.EntryContentFilter {
21  
22      public final static String DEFAULT_VENDOR_MANIFEST_ENTRY_NAME = "Implementation-Vendor-Id";
23  
24      private final String vendorId;
25      private String manifestVendorEntry = DEFAULT_VENDOR_MANIFEST_ENTRY_NAME;
26      //todo: use own events!
27      private EventSink<ContentIntrospector.Events> events = EventSink.create(ContentIntrospector.Events.class);
28  
29      public VendorFilter(String vendorId) {
30          this.vendorId = vendorId;
31      }
32  
33      public static ContentIntrospector createIntrospector(ContentIntrospector.Events listener, boolean ignoreVendorArchives, String vendorId, String manifestVendorEntry, String checkFilesPattern) {
34          final ContentIntrospector contentIntrospector = new ContentIntrospector();
35          contentIntrospector.getEvents().addListener(listener);
36          contentIntrospector.setCheckFilesPattern(checkFilesPattern);
37          if (ignoreVendorArchives) {
38              final VendorFilter vendorFilter = new VendorFilter(vendorId);
39              vendorFilter.setManifestVendorEntry(manifestVendorEntry);
40              vendorFilter.getEvents().addListener(listener);
41              contentIntrospector.setEntryContentFilter(vendorFilter);
42          }
43          return contentIntrospector;
44      }
45  
46      public void setManifestVendorEntry(String manifestVendorEntry) {
47          this.manifestVendorEntry = manifestVendorEntry;
48      }
49  
50      public EventSink<ContentIntrospector.Events> getEvents() {
51          return events;
52      }
53  
54      public boolean accept(String entryName, InputStream entryContentStream) throws IOException {
55          if (!entryName.endsWith(".jar")) return true;
56          //todo avoid the need for temporary file
57          final File tempFile = copyStreamToTemporaryFile(entryName, entryContentStream);
58          tempFile.deleteOnExit();
59          final boolean vendorArchive = checkArchiveManifest(entryName, tempFile);
60          tempFile.delete();//only for sure if the plugin is used in long live JVM
61          return !vendorArchive; // we want matching vendors to be ignored
62      }
63  
64      /**
65       * @return true when vendorId matches with jar's manifest otherwise false
66       */
67      private boolean checkArchiveManifest(final String jarPath, File tempJAR) {
68          JarFile jarFile = null;
69          try {
70              jarFile = new JarFile(tempJAR);
71              final Manifest manifest = jarFile.getManifest();
72              if(manifest != null) {
73                  final Attributes mainAttributes = manifest.getMainAttributes();
74                  if(mainAttributes != null) {
75                      final String vendor = mainAttributes.getValue(manifestVendorEntry);
76                      return vendorId.equals(vendor);
77                  }
78              }
79  
80          } catch (IOException e) {
81              events.fire.cannotCheckManifest(jarPath, e);
82          } finally {
83              if(jarFile != null) {
84                  try {
85                      jarFile.close();
86                  } catch (IOException e) {
87                      events.fire.cannotClose(jarPath, e);
88                  }
89              }
90          }
91          return false;
92      }
93  
94      private File copyStreamToTemporaryFile(final String jarPath, final InputStream archiveInputStream) throws IOException {
95          final File tempFile = File.createTempFile(UUID.randomUUID().toString(), "jar");
96          final FileOutputStream fos = new  FileOutputStream(tempFile);
97          try {
98              events.fire.checkingInTmpfile(jarPath, tempFile);
99              IOUtil.copy(archiveInputStream, fos);
100             return tempFile;
101         } finally {
102             fos.close();
103         }
104     }
