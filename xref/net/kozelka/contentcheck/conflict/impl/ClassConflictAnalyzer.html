10   * Detects class conflicts inside given set of classpath elements.
11   * @author Petr Kozelka

13  public class ClassConflictAnalyzer {

15      public ClassConflictReport analyze(Collection<ArchiveInfo> archives) {
16          final ClasspathResources cpr = new ClasspathResources();
17          for (ArchiveInfo archive : archives) {
18              for (ResourceInfo resource : archive.getResources()) {
19                  final String resourceName = resource.getUri();
20                  if (!resourceName.endsWith(".class")) continue;
21  
22                  cpr.addResource(resource, archive);
23              }
24          }
25          // prepare report
26          final ClassConflictReport report = new ClassConflictReport();
27          report.getExploredArchives().addAll(archives);
28          report.getArchiveConflicts().addAll(cpr.getConflicts());
29          report.getResources().addAll(cpr.getResources());
30          // count: involved jars, class overlaps (duplications, conflicts),
31          int totalOverlaps = 0;
32          for (ArchiveConflict archiveConflict : cpr.getConflicts()) {
33              totalOverlaps += archiveConflict.getOverlapingResources().size();
34          }
35          report.setTotalOverlaps(totalOverlaps);
36          return report;
37      }
