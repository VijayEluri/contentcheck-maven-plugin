22  public class ClassConflictDetector {
23      private final Set<ArchiveInfo> exploredArchives = new LinkedHashSet<ArchiveInfo>();
24      private final Map<String, ResourceInfo> exploredResources = new LinkedHashMap<String, ResourceInfo>();

26      private ArchiveInfo exploreArchive(ZipInputStream zis, String archiveName) throws IOException {
27  //        System.out.println("exploreArchive: " + archiveName);
28          final ArchiveInfo archiveInfo = new ArchiveInfo();
29          archiveInfo.setKey(archiveName);
30          exploredArchives.add(archiveInfo);
31          ZipEntry entry = zis.getNextEntry();
32          int classCount = 0;
33          while (entry != null) {
34              final String entryName = entry.getName();
35              if (entryName.endsWith(".class")) {
36                  processClassResource(archiveInfo, entry);
37                  classCount ++;
38              }
39              //
40              zis.closeEntry();
41              entry = zis.getNextEntry();
42          }
43          archiveInfo.setClassCount(classCount);
44          return archiveInfo;
45      }
46  
47      private void processClassResource(ArchiveInfo archiveInfo, ZipEntry entry) {
48          final String key = entry.getName();
49  //        System.out.println(" : " + key);
50          ResourceInfo resourceInfo = exploredResources.get(key);
51          if (resourceInfo == null) {
52              resourceInfo = new ResourceInfo();
53              resourceInfo.key = key;
54              exploredResources.put(key, resourceInfo);
55          }
56          resourceInfo.addHostingArchive(archiveInfo);
57      }

59      public Set<ArchiveInfo> getExploredArchives() {
60          return exploredArchives;
61      }
62  
63      public List<ArchiveInfo> getConflictingArchives() {
64          final List<ArchiveInfo> conflictingArchives = new ArrayList<ArchiveInfo>();
65          for (ArchiveInfo archive : exploredArchives) {
66              if (! archive.getConflicts().isEmpty()) {
67                  conflictingArchives.add(archive);
68              }
69          }
70          return conflictingArchives;
71      }

73      public static ClassConflictDetector exploreWar(File war) throws IOException {
74          final ClassConflictDetector ccd = new ClassConflictDetector();
75          final ZipInputStream waris = new ZipInputStream(new FileInputStream(war));
76          ZipEntry entry = waris.getNextEntry();
77          while (entry != null) {
78              final String entryName = entry.getName();
79              if (entryName.startsWith("WEB-INF/lib/") && entryName.endsWith(".jar")) {
80                  final ZipInputStream zis = new ZipInputStream(waris);
81                  ccd.exploreArchive(zis, entry.getName());
82              }
83              waris.closeEntry();
84              entry = waris.getNextEntry();
85          }
86          return ccd;
87      }

89      public int printResults(int previewThreshold, LineOutput output) {
90          final List<ArchiveInfo> sortedConflictingArchives = new ArrayList<ArchiveInfo>(getConflictingArchives());
91          Collections.sort(sortedConflictingArchives, new Comparator<ArchiveInfo>() {
92              public int compare(ArchiveInfo o1, ArchiveInfo o2) {
93                  return o1.getKey().compareTo(o2.getKey());
94              }
95          });
96          int totalConflicts = 0;
97          for (ArchiveInfo cai : sortedConflictingArchives) {
98              output.println("");
99              output.println(String.format("File '%s' (%d classes):", cai.getKey(), cai.getClassCount()));
100             final List<Conflict> sortedConflicts = new ArrayList<Conflict>(cai.getConflicts());
101             Collections.sort(sortedConflicts, new Comparator<Conflict>() {
102                 public int compare(Conflict o1, Conflict o2) {
103                     return o1.getOtherArchive().getKey().compareTo(o2.getOtherArchive().getKey());
104                 }
105             });
106             for (Conflict conflict : sortedConflicts) {
107                 final List<ResourceInfo> conflictResources = conflict.getResources();
108                 final int conflictResourceCount = conflictResources.size();
109                 totalConflicts += conflictResourceCount;
110                 output.println(String.format("%8d class conflicts with '%s'", conflictResourceCount, conflict.getOtherArchive().getKey()));
111                 if (previewThreshold == 0) continue;
112                 int cnt = 0;
113                 for (ResourceInfo resource : conflictResources) {
114                     cnt ++;
115                     if (previewThreshold >= 0) {
116                         if (cnt > previewThreshold) {
117                             output.println("                ...");
118                             break;
119                         }
120                     }
121                     output.println(String.format("                %s", resource.key));
122                 }
123             }
124         }
125         output.println("-------------------------------------------------");
126         output.println(String.format("Total: %d conflicts affect %d of %d archives.",
127                 totalConflicts,
128                 sortedConflictingArchives.size(),
129                 getExploredArchives().size()));
130         return totalConflicts;
131     }
