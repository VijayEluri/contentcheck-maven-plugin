14  /**
15   * Utilities helping to load archive information from the filesystem
16   *
17   * @author Petr Kozelka
18   */
19  public class ArchiveLoader {
20      public static List<ArchiveInfo> loadWar(File war) throws IOException {
21          final List<ArchiveInfo> archives = new ArrayList<ArchiveInfo>();
22          final ContentIntrospector ci = new ContentIntrospector();
23          ci.setSourceFile(war);
24          ci.setEntryContentFilter(new ContentIntrospector.EntryContentFilter() {
25              @Override
26              public boolean accept(String entryName, InputStream entryContentStream) throws IOException {
27                  if (entryName.startsWith("WEB-INF/lib/") && entryName.endsWith(".jar")) {
28                      final ZipInputStream zis = new ZipInputStream(entryContentStream);
29                      archives.add(loadInnerArchive(zis, entryName));
30                  }
31                  //TODO: add support for WEB-INF/classes as another resource
32                  return false;
33              }
34          });
35          ci.walk();
36          return archives;
37      }
38  
39      private static ArchiveInfo loadInnerArchive(ZipInputStream zis, String archiveName) throws IOException {
40          final ArchiveInfo archive = new ArchiveInfo();
41          archive.setKey(archiveName);
42          ZipEntry entry = zis.getNextEntry();
43          while (entry != null) {
44              if (!entry.isDirectory()) {
45                  processResource(archive, entry);
46              }
47              //
48              zis.closeEntry();
49              entry = zis.getNextEntry();
50          }
51          return archive;
52      }
53  
54      private static void processResource(ArchiveInfo archive, ZipEntry entry) {
55          final ResourceInfo resource = new ResourceInfo();
56          resource.setUri(entry.getName());
57          resource.setHash(String.format("crc=%s&sz=%s", entry.getCrc(), entry.getSize()));
58          archive.addResource(resource);
59      }
